```sql
SELECT B.DNAME, SUM(A.SAL)
FROM DEPT B, EMP A
WHERE B.DEPTNO = A.DEPTNO
GROUP BY B.DNAME
UNION ALL
SELECT '', SUM(A.SAL)
FROM EMP A
ORDER BY 1
;
```
