```SQL
SELECT JOB
	, MAX(SAL) AS SAL_MAX
	, MIN(SAL) AS SAL_MIN
	, SUM(SAL) AS SAL_TOTAL
	, ROUND(AVG(SAL),0) AS SAL_AVG
FROM EMP 
GROUP BY JOB 
HAVING SUM(SAL) >= 6000
;
```

```SQL
SELECT DEPTNO
	, COUNT(DEPTNO) AS DEPT_EMP_CNT
```
